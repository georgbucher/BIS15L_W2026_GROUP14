Ambient Temperature

egg_laying %>%
  select(temperature_ambient, n_eggs) %>% 
  mutate(lay_eggs = if_else(n_eggs > 0, "Yes", "No")) %>%
  ggplot(aes(x = lay_eggs, y = temperature_ambient, fill = lay_eggs)) +
  geom_boxplot() +
  scale_fill_manual(values = c("Yes" = "lightblue", "No" = "orange")) +
  coord_flip() +
  theme_minimal() +
  theme(legend.position = "none") +
  labs(
    title = "Egg Presence based on Ambient Temperature",
    x = "Any Eggs?",
    y = "Ambient Temperature (°C)"
    )

egg_laying %>%
  filter(n_eggs != 0) %>%
  ggplot(aes(x = temperature_ambient, y = n_eggs)) +
  geom_jitter(alpha = 0.4) +
  geom_smooth(method = "lm", se = FALSE, color = "red") +
  theme_minimal() +
  labs(
    title = "Amount of Butterfly Eggs Layed based on Ambient Temperature",
    x = "Ambient Temperature (°C)",
    y = "Amount of Eggs"
  )
## `geom_smooth()` using formula = 'y ~ x'

egg_laying %>%
  filter(n_eggs != 0) %>%
  ggplot(aes(x = temperature_ambient)) +
  geom_histogram(aes(y = after_stat(density)), color = "black", fill = "lightgreen") +
  geom_density(color = "red", alpha = 0.4, size = 1.5) +
  labs(
    title = "Distribution of Ambient Temperatures for Plants with Butterfly Eggs",
    x = "Plant Nitrogen Content (Mmol/g)",
    y = "Count"
  )
